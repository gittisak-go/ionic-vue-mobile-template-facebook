3.3 Vuex Store Setup (src/store/index.js)

   import { createStore } from 'vuex';

   export default createStore({
     state: {
       posts: [
         // Seeded posts data
       ],
       stories: [
         // Seeded stories data
       ],
       user: {
         name: 'Current User',
         avatar: 'assets/images/avatar.png'
       },
       chatMessages: []
     },
     
     getters: {
       posts: (state) => state.posts,
       stories: (state) => state.stories,
       user: (state) => state.user
     },
     
     mutations: {
       ADD_POST(state, post) {
         state.posts.unshift(post);
       },
       ADD_CHAT_MESSAGE(state, message) {
         state.chatMessages.push(message);
       }
     },
     
     actions: {
       addPost({ commit }, post) {
         commit('ADD_POST', post);
       }
     }
   });

   const sharp = require('sharp');
   const fs = require('fs');
   const path = require('path');
   
   const imagesDir = path.join(__dirname, 'public', 'assets', 'images');
   
   async function optimizeImage(filePath) {
     try {
       const fileName = path.basename(filePath);
       const ext = path.extname(filePath).toLowerCase();
       
       // ‡∏Ç‡πâ‡∏≤‡∏° favicon
       if (fileName === 'favicon.png') {
         console.log(`‚è≠Ô∏è  Skipping ${fileName}`);
         return;
       }
       
       // ‡∏≠‡πà‡∏≤‡∏ô‡∏Ç‡πâ‡∏≠‡∏°‡∏π‡∏•‡πÄ‡∏î‡∏¥‡∏°
       const originalSize = fs.statSync(filePath).size;
       const tempPath = filePath + '.tmp';
       
       // Optimize ‡∏ï‡∏≤‡∏°‡∏ä‡∏ô‡∏¥‡∏î‡πÑ‡∏ü‡∏•‡πå
       if (ext === '.jpg' || ext === '.jpeg') {
         await sharp(filePath)
           .jpeg({
             quality: 75,
             progressive: true,
             mozjpeg: true
           })
           .toFile(tempPath);
       } else if (ext === '.png') {
         await sharp(filePath)
           .png({
             quality: 80,
             compressionLevel: 9,
             progressive: true
           })
           .toFile(tempPath);
       } else {
         console.log(`‚è≠Ô∏è  Skipping ${fileName} (unsupported format)`);
         return;
       }
       
       // ‡πÄ‡∏õ‡∏£‡∏µ‡∏¢‡∏ö‡πÄ‡∏ó‡∏µ‡∏¢‡∏ö‡∏Ç‡∏ô‡∏≤‡∏î
       const optimizedSize = fs.statSync(tempPath).size;
       
       if (optimizedSize < originalSize) {
         // Replace ‡∏ñ‡πâ‡∏≤‡πÄ‡∏•‡πá‡∏Å‡∏Å‡∏ß‡πà‡∏≤
         fs.unlinkSync(filePath);
         fs.renameSync(tempPath, filePath);
         
         const saved = originalSize - optimizedSize;
         const percent = ((saved / originalSize) * 100).toFixed(1);
         
         console.log(`‚úÖ ${fileName}`);
         console.log(`   ${(originalSize / 1024).toFixed(0)}KB ‚Üí ${(optimizedSize / 1024).toFixed(0)}KB (${percent}% reduction)`);
       } else {
         // ‡∏ñ‡πâ‡∏≤‡πÑ‡∏°‡πà‡πÄ‡∏•‡πá‡∏Å‡∏Å‡∏ß‡πà‡∏≤ ‡πÉ‡∏ä‡πâ‡πÑ‡∏ü‡∏•‡πå‡πÄ‡∏î‡∏¥‡∏°
         fs.unlinkSync(tempPath);
         console.log(`‚ö†Ô∏è  ${fileName} - Original is smaller, keeping it`);
       }
       
     } catch (error) {
       console.error(`‚ùå Error optimizing ${filePath}:`, error.message);
     }
   }
   
   async function optimizeAllImages() {
     console.log('üñºÔ∏è  Starting image optimization...\n');
     
     const files = fs.readdirSync(imagesDir);
     const imageFiles = files.filter(file => {
       const ext = path.extname(file).toLowerCase();
       return ['.jpg', '.jpeg', '.png'].includes(ext);
     });
     
     let totalOriginal = 0;
     let totalOptimized = 0;
     
     for (const file of imageFiles) {
       const filePath = path.join(imagesDir, file);
       const sizeBefore = fs.statSync(filePath).size;
       
       await optimizeImage(filePath);
       
       const sizeAfter = fs.statSync(filePath).size;
       totalOriginal += sizeBefore;
       totalOptimized += sizeAfter;
     }
     
     const totalSaved = totalOriginal - totalOptimized;
     const totalPercent = ((totalSaved / totalOriginal) * 100).toFixed(1);
     
     console.log('\nüìä Summary:');
     console.log(`   Total original: ${(totalOriginal / 1024 / 1024).toFixed(2)} MB`);
     console.log(`   Total optimized: ${(totalOptimized / 1024 / 1024).toFixed(2)} MB`);
     console.log(`   Total saved: ${(totalSaved / 1024 / 1024).toFixed(2)} MB (${totalPercent}%)`);
     console.log(`   Images processed: ${imageFiles.length}`);
   }
   
   optimizeAllImages();
